123     public function testCharset()
124     {
125         if (!extension_loaded('mbstring')) {
126             $this->markTestSkipped('This test requires mbstring.');
127         }
128         $var = mb_convert_encoding('Словарь', 'CP1251', 'UTF-8');
129 
130         $dumper = new HtmlDumper('php://output', 'CP1251');
131         $dumper->setDumpHeader('<foo></foo>');
132         $dumper->setDumpBoundaries('<bar>', '</bar>');
133         $cloner = new VarCloner();
134 
135         $data = $cloner->cloneVar($var);
136         $out = fopen('php://memory', 'r+b');
137         $dumper->dump($data, $out);
138         rewind($out);
139         $out = stream_get_contents($out);
140 
141         $this->assertStringMatchesFormat(
142             <<<EOTXT
143 <foo></foo><bar>b"<span class=sf-dump-str title="7 binary or non-UTF-8 characters">&#1057;&#1083;&#1086;&#1074;&#1072;&#1088;&#1100;</span>"
144 </bar>
145 
146 EOTXT
147             ,

149             $out
150         );
