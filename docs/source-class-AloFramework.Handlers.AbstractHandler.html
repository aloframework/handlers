161         /**
162          * Formats the debug backtrace row
163          * @author Art <a.molcanovas@gmail.com>
164          *
165          * @param array $traceLine The row
166          * @param mixed $method    Reference to the variable which will contain the formatted method
167          * @param mixed $file      Reference to the variable which will contain the formatted file location
168          * @param mixed $line      Reference to the variable which will contain the formatted line
169          */
170         private static function formatTraceLine(array $traceLine, &$method, &$file, &$line) {
171             $method = $file = $line = '';
172 
173             if (isset($traceLine['class'])) {
174                 $method = $traceLine['class'];
175             }
176             if (isset($traceLine['type'])) {
177                 $method .= $traceLine['type'];
178             }
179             if (isset($traceLine['function'])) {
180                 $method .= $traceLine['function'] . '()';
181             }
182             if (!$method) {
183                 $method = '[unknown]';
184             }
185 
186             if (isset($traceLine['file'])) {
187                 $file = implode(DIRECTORY_SEPARATOR, array_slice(explode(DIRECTORY_SEPARATOR, $traceLine['file']), -2));
188             }
189 
190             if (array_key_exists('line', $traceLine)) {
191                 $line = $traceLine['line'];
192             }
193         }

195         /**
196          * CLI output of the debug backtrace
197          * @author Art <a.molcanovas@gmail.com>
198          *
199          * @param array  $trace The debug backtrace
200          * @param string $label Colour id
201          */
202         private function traceCLI(array $trace, $label) {
203             foreach ($trace as $k => $v) {
204                 $func        = $loc = $line = '';
205                 $argsPresent = isset($v['args']) && !empty($v['args']);
206 
207                 self::formatTraceLine($v, $func, $loc, $line);
208 
209                 $this->console->write('<' . $label . 'b>#: </>')
210                     ->write('<' . $label . '>' . $k . '</>', true)
211                     ->write('<' . $label . 'b>Method: </>')
212                     ->write('<' . $label . '>' . $func . '</>', true)
213                     ->write('<' . $label . 'b>Args:</>', $argsPresent);
214 
215                 if ($argsPresent) {
216                     VarDumper::dump($v['args']);
217                 } else {
218                     $this->console->write('<' . $label . '> [none]</>', true);
219                 }
220 
221                 $this->console->write('<' . $label . 'b>File: </>')
222                     ->write('<' . $label . '>' . ($loc ? $loc : '<<unknown>>') . '</>',
223                             true)
224                     ->write('<' . $label . 'b>Line: </>')
225                     ->write('<' . $label . '>' . ($line ? $line : '<<unknown>>') . '</>', true)
226                     ->write('', true);
227             }
228         }
