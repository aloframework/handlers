105             foreach ($trace as $k => $v) {
106                 $func = $loc = $line = '';
107 
108                 if (isset($v['class'])) {
109                     $func = $v['class'];
110                 }

114                 if (isset($v['function'])) {

121                 if (isset($v['file'])) {

123                 }
124                 if (isset($v['line'])) {
125                     $line .= $v['line'];
126                 }
127 
128                 echo '<tr>'
129                      //BEGIN row
130                      .
131                      '<td>' .
132                      $k .
133                      '</td>'
134                      //Trace #
135                      .
136                      '<td>' .
137                      $func .
138                      '</td>'
139                      //Method used
140                      .
141                      '<td>' .

144                          '<span class="label label-default">NONE</span>') .

146                      //END args
147                      .

159                      '</td>'

161                      .
162                      '</tr>';
163             }

165             echo '</tbody>' . '</table>';
