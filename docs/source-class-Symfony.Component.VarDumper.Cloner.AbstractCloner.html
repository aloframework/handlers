123     /**
124      * Adds casters for resources and objects.
125      *
126      * Maps resources or objects types to a callback.
127      * Types are in the key, with a callable caster for value.
128      * Resource types are to be prefixed with a `:`,
129      * see e.g. static::$defaultCasters.
130      *
131      * @param callable[] $casters A map of casters.
132      */
133     public function addCasters(array $casters)
134     {
135         foreach ($casters as $type => $callback) {
136             $this->casters[strtolower($type)][] = $callback;
137         }
138     }

160     /**
161      * Clones a PHP variable.
162      *
163      * @param mixed $var    Any PHP variable.
164      * @param int   $filter A bit field of Caster::EXCLUDE_* constants.
165      *
166      * @return Data The cloned variable represented by a Data object.
167      */
168     public function cloneVar($var, $filter = 0)
169     {
170         $this->filter = $filter;
171         $this->prevErrorHandler = set_error_handler(array($this, 'handleError'));
172         try {
173             if (!function_exists('iconv')) {
174                 $this->maxString = -1;
175             }
176             $data = $this->doClone($var);
177         } catch (\Exception $e) {
178         }
179         restore_error_handler();
180         $this->prevErrorHandler = null;
181 
182         if (isset($e)) {
183             throw $e;
184         }
185 
186         return new Data($data);
187     }

198     /**
199      * Casts an object to an array representation.
200      *
201      * @param Stub $stub     The Stub for the casted object.
202      * @param bool $isNested True if the object is nested in the dumped structure.
203      *
204      * @return array The object casted as array.
205      */
206     protected function castObject(Stub $stub, $isNested)
207     {
208         $obj = $stub->value;
209         $class = $stub->class;
210 
211         if (isset($this->classInfo[$class])) {
212             $classInfo = $this->classInfo[$class];
213             $stub->class = $classInfo[0];
214         } else {
215             $classInfo = array(
216                 $class,
217                 new \ReflectionClass($class),
218                 array_reverse(array($class => $class) + class_parents($class) + class_implements($class) + array('*' => '*')),
219             );
220 
221             $this->classInfo[$class] = $classInfo;
222         }
223 
224         $a = $this->callCaster('Symfony\Component\VarDumper\Caster\Caster::castObject', $obj, $classInfo[1], null, $isNested);
225 
226         foreach ($classInfo[2] as $p) {
227             if (!empty($this->casters[$p = strtolower($p)])) {
228                 foreach ($this->casters[$p] as $p) {
229                     $a = $this->callCaster($p, $obj, $a, $stub, $isNested);
230                 }
231             }
232         }
233 
234         return $a;
235     }

237     /**
238      * Casts a resource to an array representation.
239      *
240      * @param Stub $stub     The Stub for the casted resource.
241      * @param bool $isNested True if the object is nested in the dumped structure.
242      *
243      * @return array The resource casted as array.
244      */
245     protected function castResource(Stub $stub, $isNested)
246     {
247         $a = array();
248         $res = $stub->value;
249         $type = $stub->class;
250 
251         if (!empty($this->casters[':'.$type])) {
252             foreach ($this->casters[':'.$type] as $c) {
253                 $a = $this->callCaster($c, $res, $a, $stub, $isNested);
254             }
255         }
256 
257         return $a;
258     }

260     /**
261      * Calls a custom caster.
262      *
263      * @param callable        $callback The caster.
264      * @param object|resource $obj      The object/resource being casted.
265      * @param array           $a        The result of the previous cast for chained casters.
266      * @param Stub            $stub     The Stub for the casted object/resource.
267      * @param bool            $isNested True if $obj is nested in the dumped structure.
268      *
269      * @return array The casted object/resource.
270      */
271     private function callCaster($callback, $obj, $a, $stub, $isNested)
272     {
273         try {
274             $cast = call_user_func($callback, $obj, $a, $stub, $isNested, $this->filter);
275 
276             if (is_array($cast)) {
277                 $a = $cast;
278             }
279         } catch (\Exception $e) {
280             $a[(Stub::TYPE_OBJECT === $stub->type ? Caster::PREFIX_VIRTUAL : '').'⚠'] = new ThrowingCasterException($callback, $e);
281         }
282 
283         return $a;
284     }

286     /**
287      * Special handling for errors: cloning must be fail-safe.
288      *
289      * @internal
290      */
291     public function handleError($type, $msg, $file, $line, $context)
292     {
293         if (E_RECOVERABLE_ERROR === $type || E_USER_ERROR === $type) {
294             // Cloner never dies
295             throw new \ErrorException($msg, 0, $type, $file, $line);
296         }
297 
298         if ($this->prevErrorHandler) {
299             return call_user_func($this->prevErrorHandler, $type, $msg, $file, $line, $context);
300         }
301 
302         return false;
303     }
