 28         /**
 29          * The error handler
 30          *
 31          * @author Art <a.molcanovas@gmail.com>
 32          *
 33          * @param int    $errno   The level of the error raised
 34          * @param string $errstr  The error message
 35          * @param string $errfile The filename that the error was raised in
 36          * @param int    $errline The line number the error was raised at
 37          */
 38         function handle($errno, $errstr, $errfile, $errline) {
 39             $this->injectCss();
 40             $type  = $errno;
 41             $label = 'warning';
 42 
 43             switch ($errno) {
 44                 case E_NOTICE:
 45                 case E_USER_NOTICE:
 46                     $type  = 'NOTICE';
 47                     $label = 'info';
 48                     break;
 49                 case E_ERROR:
 50                 case E_USER_ERROR:
 51                 case E_COMPILE_ERROR:
 52                 case E_RECOVERABLE_ERROR:
 53                 case E_CORE_ERROR:
 54                     $type  = 'ERROR';
 55                     $label = 'danger';
 56                     break;
 57                 case E_DEPRECATED:
 58                 case E_USER_DEPRECATED:
 59                     $type = 'DEPRECATED';
 60                     break;
 61                 case E_WARNING:
 62                 case E_USER_WARNING:
 63                 case E_CORE_WARNING:
 64                     $type = 'WARNING';
 65                     break;
 66             }
 67 
 68             $file = implode(DIRECTORY_SEPARATOR, array_slice(explode(DIRECTORY_SEPARATOR, $errfile), -2));
 69 
 70             if (!$errline) {
 71                 $errline = '<<unknown>>';
 72             }
 73 
 74             if ($this->isCLI) {
 75                 $label = $label == 'danger' ? 'e' : substr($label, 0, 1);
 76                 $this->handleCLI($type, $label, $errno, $errstr, $file, $errline);
 77             } else {
 78                 $this->handleHTML($type, $label, $errno, $errstr, $file, $errline);
 79             }
 80 
 81             $this->log($errno, $errstr);
 82         }
 83

 92         protected function handleHTML($type, $label, $errno, $errstr, $errfile, $errline) {

100                         <span>[<?= $errno ?>] <?= $errstr ?></span>
101                     </div>
102                     <div>
103                         <span class="alo-bold">Raised in </span>
104                         <span class="alo-uline"><?= $errfile ?></span>
105                         <span> @ line </span>
106                         <span class="alo-uline"><?= $errline ?></span>
107                     </div>
108                     <div>
109                         <span class="alo-bold">Backtrace:</span>
110                         <?= $this->getTrace($trace, $label) ?>
111                     </div>
112                 </div>
113             </div>
114             <?php

126         protected function handleCLI($type, $label, $errno, $errstr, $errfile, $errline) {
127             $this->console->write('<' . $label . 'b>' . $type . '</>')
128                 ->write('<' . $label . '>: [' . $errno . '] ' . $errstr . '</>',
129                         true)
130                 ->write('<' . $label . '>Raised in </>')
131                 ->write('<' . $label . 'u>' . $errfile . '</>')
132                 ->write('<' . $label . '> @ line </><' . $label . 'u>' . $errline . '</>', true)
133                 ->write('<' . $label . 'b>Debug backtrace:</>', true)
134                 ->writeln('');
135 
136             $this->getTrace(array_slice(debug_backtrace(), 2), $label);
137         }
